使用`Object.defineProperty`方法为对象创建属性
在获取属性值或者赋值时会自动调用其中的get和set方法

```
var person = {
        age: 16
    }
    Object.defineProperty(person, 'userName', {
        //value: '张三',
        get: function () {
            console.log('获取属性值时自动触发')
        },
        set: function (value) {
            console.log('为属性赋值时自动触发:'+value)
        }
    })
    person.userName = '张三'
    person.userName
```
